// TERMINALS
actor BaseTerminal {
	+FULLVOLACTIVE
	+SHOOTABLE
	+NOBLOOD
	+DONTGIB
	+NOICEDEATH
	+QUICKTORETALIATE
	+NEVERRESPAWN
	+GHOST
	
	Height 120
	Mass 0x7FFFFFFF
	Speed 100 // used for base hud health percentages! DON'T TOUCH IT
	DamageType "Explosion"
	Obituary "%o went down with a \cGRed\c- building."
	Radius 16
	DeathSound "grenade/explode"
	PainChance 255
	DesignatedTeam 1
	
	DamageFactor "Fist", 0.2
	DamageFactor "Bullet", 0.5
	DamageFactor "Rifle", 0.5
	DamageFactor "Explosion", 0.75
	DamageFactor "Fire", 0.6
	DamageFactor "FireSpecial", 0.0
	DamageFactor "Chemical", 0.75
	DamageFactor "ChemicalSpecial", 0.0
	DamageFactor "Laser", 1.0
	DamageFactor "RazorWire", 0.0
	DamageFactor "AntiArmor", 0.25
	DamageFactor "Time", 1.0
	DamageFactor "Disarm", 0.0
	DamageFactor "Superweapon", 1.0
	DamageFactor "MassDriver", 0.25
	DamageFactor "Mechstomp", 0.0
	DamageFactor "Monster", 0.0
	DamageFactor "Deconstruction", 0.0
	
	States {
	Spawn:
		TNT1 A 0
		TNT1 A 0 A_GiveInventory ("IsMCT", 1)
		TNT1 A -1
		stop
	
	Idle:
		TNT1 A -1
		stop
	
	SpawnDone:
		TNT1 A 0
		TNT1 A 0 A_ChangeFlag ("ISMONSTER", 1)
		TNT1 A 0 A_ClearTarget
		TNT1 A -1 A_Stop
		stop
	
	Pain:
		goto Spawn
	
	Pain.Chemical:
	Pain.Fire:
		goto Pain+1
	
	Death:
		TNT1 A 5 A_Scream
		TNT1 A 5 A_Scream
		TNT1 A 0 bright A_PlayWeaponSound("weapons/rocklx")
		TNT1 A 0 A_SpawnItemEx ("TerminalExplosion")
		TNT1 A 0 A_SpawnItemEx ("ExplosionMainLarge")
		TNT1 A 5 A_Scream
		TNT1 A 0 bright A_Scream
		TNT1 A 0 A_SpawnItemEx ("TerminalExplosion")
		TNT1 A 0 bright A_PlayWeaponSound("weapons/rocklx")
		TNT1 A 0 A_SpawnItemEx ("TerminalExplosion")
		TNT1 A 0 A_SpawnItemEx("ExplosionMainLarge")
		TNT1 A 5 A_Scream
		TNT1 A 5 A_Scream
		TNT1 A 0 bright A_PlayWeaponSound("weapons/rocklx")
		TNT1 A 0 A_SpawnItemEx ("TerminalExplosion")
		TNT1 A 0 A_SpawnItemEx ("ExplosionMainLarge")
		TNT1 A 5 A_Scream
		TNT1 A 5 A_Scream
		TNT1 A 0 bright A_Scream
		TNT1 A 5 
		TNT1 A -1 A_Scream
		stop
	}
}
